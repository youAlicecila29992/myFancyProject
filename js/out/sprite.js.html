'use strict';

class Sprite{

    /**
     * Sprite konstruieren
     * @param pfad zum bild
     */
    constructor(name){
        this.img=new Image();
        this.img.src=name;
    }

    /**
     * Quellbild holen
     * @return liefert das Bild zur√ºck
     */ 
    getImage(){
        return this.img;
    }

    /**Quellbild setzten
     * @param Image Setzt ein neues Bild
     */
    setImage(iimage){
        img=iimage;
    }

    /**
     * In ein graphisches Element malen
     * @param cont 2d-Context des Elements
     * @param x Position x
     * @param y Position y
     */ 
    draw(cont,x,y){
        cont.draw(Image,x,y);
    }

}
